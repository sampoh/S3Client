< S3 APIの通信設定 >  
  
第1引数 : 設定を格納したオブジェクト 【 オブジェクト型 】  
戻り値 : 反映後の設定値 【 オブジェクト型 】  
  
[ 設定可能な項目 ( いずれも任意 ) ]  
　{  
　　"partSize":< アップロード時の分割サイズ(バイト数) ( 数値 ) >,  
　　"upload":{  
　　　"maxRetry":< アップロード処理全体を通したリトライ上限回数 ( 数値 ) >,  
　　　"maxRetryEach":< アップロード処理の個別通信処理のリトライ上限回数 ( 数値 ) >,  
　　　"retryWait":< リトライ時の待機時間(秒) ( 数値 ) >  
　　}  
　}  
  
[ デフォルト値 ]  
　{  
　　"partSize":5242880,  
　　"upload":{  
　　　"maxRetry":10,  
　　　"maxRetryEach":5,  
　　　"retryWait":3  
　　}  
　}
